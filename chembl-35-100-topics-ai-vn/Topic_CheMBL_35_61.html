def calculate_descriptors(smiles): try: mol = Chem.MolFromSmiles(smiles) if mol is None: return None descriptors = {} for name, func in Descriptors.descList: try: descriptors[name] = func(mol) except: descriptors[name] = np.nan return pd.Series(descriptors) except: return None
