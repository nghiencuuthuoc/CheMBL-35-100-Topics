```python import pandas as pd from rdkit import Chem from rdkit.Chem import Descriptors from sklearn.model_selection import train_test_split from sklearn.ensemble import RandomForestRegressor from sklearn.metrics import mean_squared_error, r2_score import numpy as np import os

1. T√≠nh LogP b·∫±ng RDKit

def calculate_morgan_fingerprint(smiles, radius=2, nBits=2048): try: mol = Chem.MolFromSmiles(smiles) if mol: fp = AllChem.GetMorganFingerprintAsBitVect(mol, radius, nBits=nBits) return np.array(list(fp.ToBitString()), dtype=int) else: return None except: return None

df['morgan_fp'] = df['canonical_smiles'].apply(calculate_morgan_fingerprint) df = df.dropna(subset=['morgan_fp'])
